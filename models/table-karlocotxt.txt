CREATE TABLE `karloco`.`user` ( `id_user` INT NOT NULL AUTO_INCREMENT , `nom` VARCHAR(50) NOT NULL , `prenom` VARCHAR(50) NOT NULL , `email` VARCHAR(50) NOT NULL , `sexe` VARCHAR(2) NOT NULL , `tel` VARCHAR(20) NOT NULL , `ville` VARCHAR(50) NOT NULL , `province` VARCHAR(50) NOT NULL , `password` VARCHAR(255) NOT NULL , `categorie` VARCHAR(20) NOT NULL , PRIMARY KEY (`id_user`)) ENGINE = MyISAM; 
CREATE TABLE `karloco`.`voiture` ( `immatriculation` VARCHAR(20) NOT NULL , `id_user` INT NOT NULL , `marque` VARCHAR(30) NOT NULL , `tarif` INT NOT NULL , `couleur` VARCHAR(20) NOT NULL , `nbplace` INT NOT NULL , `image` TEXT NOT NULL , `createdat` DATE NOT NULL , PRIMARY KEY (`immatriculation`)) ENGINE = MyISAM; 
CREATE TABLE `karloco`.`reservation` ( `id_res` INT NOT NULL AUTO_INCREMENT , `id_user` INT NOT NULL , `immatriculation` VARCHAR(20) NOT NULL , `nbjour` INT NOT NULL , `dateres` DATE NOT NULL , `lieu` VARCHAR(50) NOT NULL , `heure` VARCHAR(10) NOT NULL , PRIMARY KEY (`id_res`)) ENGINE = MyISAM; 
CREATE TABLE `karloco`.`favoris` ( `id_fav` INT NOT NULL AUTO_INCREMENT , `id_user` INT NOT NULL , `immatriculation` VARCHAR(20) NOT NULL , PRIMARY KEY (`id_fav`)) ENGINE = MyISAM; 
CREATE TABLE `karloco`.`notification` ( `id_notif` INT NOT NULL AUTO_INCREMENT , `sender` INT NOT NULL , `receiver` INT NOT NULL , `content` TEXT NOT NULL , `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `lu` BOOLEAN NOT NULL DEFAULT FALSE , PRIMARY KEY (`id_notif`)) ENGINE = MyISAM; 
CREATE TABLE `karloco`.`conversation` ( `id_conv` VARCHAR(20) NOT NULL , `user1` INT NOT NULL , `user2` INT NOT NULL , `createdAt` DATETIME NOT NULL ), PRIMARY KEY(`id_conv`) ENGINE = MyISAM; 
CREATE TABLE `karloco`.`message` ( `id_mes` INT NOT NULL AUTO_INCREMENT , `id_conv` VARCHAR(20) NOT NULL , `sender` INT NOT NULL , `text` TEXT NOT NULL , `datemes` DATETIME NOT NULL , `lu` BOOLEAN NOT NULL DEFAULT FALSE , `last` BOOLEAN NOT NULL DEFAULT TRUE , PRIMARY KEY (`id_mes`)) ENGINE = MyISAM;
